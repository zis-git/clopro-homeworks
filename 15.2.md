### Файл `providers.tf`

```hcl
terraform {
  required_version = ">= 1.5.0"

  required_providers {
    yandex = {
      source  = "yandex-cloud/yandex"
      version = ">= 0.108.0"
    }
  }
}

provider "yandex" {
  cloud_id  = var.cloud_id
  folder_id = var.folder_id
  zone      = var.zone
}
```

### Файл `main.tf`

```hcl
resource "yandex_vpc_network" "develop" {
  name = var.network_name
}

resource "yandex_vpc_subnet" "public" {
  name           = var.subnet_name
  zone           = var.zone
  network_id     = yandex_vpc_network.develop.id
  v4_cidr_blocks = var.public_cidr
}

resource "yandex_iam_service_account" "ig_sa" {
  name = "lamp-group-sa"
}

resource "yandex_resourcemanager_folder_iam_member" "ig_editor" {
  folder_id = var.folder_id
  role      = "editor"
  member    = "serviceAccount:${yandex_iam_service_account.ig_sa.id}"
}

resource "yandex_storage_bucket" "site_bucket" {
  bucket        = var.bucket_name
  force_destroy = true

  anonymous_access_flags {
    read        = true
    list        = false
    config_read = true
  }
}

resource "yandex_storage_object" "image" {
  bucket = yandex_storage_bucket.site_bucket.bucket
  key    = var.image_name
  source = "${path.module}/${var.image_name}"
}

resource "yandex_compute_instance_group" "lamp_group" {
  name                = "lamp-group"
  folder_id           = var.folder_id
  service_account_id  = yandex_iam_service_account.ig_sa.id
  deletion_protection = false

  instance_template {
    platform_id = "standard-v1"

    resources {
      cores  = 2
      memory = 2
    }

    boot_disk {
      initialize_params {
        image_id = "fd827b91d99psvq5fjit"
        size     = 10
        type     = "network-hdd"
      }
    }

    network_interface {
      network_id = yandex_vpc_network.develop.id
      subnet_ids = [yandex_vpc_subnet.public.id]
      nat        = true
    }

    metadata = {
      user-data = <<-EOF
        #cloud-config
        write_files:
          - path: /var/www/html/index.html
            permissions: '0644'
            content: |
              <!doctype html>
              <html lang="ru">
              <head>
                <meta charset="UTF-8">
                <title>Instance Group works</title>
              </head>
              <body>
                <h1>Instance Group works</h1>
                <p>Изображение из Object Storage:</p>
                <img src="https://storage.yandexcloud.net/${yandex_storage_bucket.site_bucket.bucket}/${yandex_storage_object.image.key}" alt="image from bucket" width="320">
              </body>
              </html>
        runcmd:
          - systemctl restart apache2 || systemctl restart httpd || true
      EOF
    }
  }

  scale_policy {
    fixed_scale {
      size = 3
    }
  }

  allocation_policy {
    zones = [var.zone]
  }

  deploy_policy {
    max_unavailable = 1
    max_expansion   = 0
  }

  health_check {
    interval            = 10
    timeout             = 5
    healthy_threshold   = 2
    unhealthy_threshold = 2

    http_options {
      port = 80
      path = "/"
    }
  }

  load_balancer {
    target_group_name = "lamp-target-group"
  }
}

resource "yandex_lb_network_load_balancer" "lamp_nlb" {
  name = "lamp-network-load-balancer"

  listener {
    name        = "http-listener"
    port        = 80
    target_port = 80

    external_address_spec {
      ip_version = "ipv4"
    }
  }

  attached_target_group {
    target_group_id = yandex_compute_instance_group.lamp_group.load_balancer[0].target_group_id

    healthcheck {
      name = "http-healthcheck"

      http_options {
        port = 80
        path = "/"
      }
    }
  }
}
```

### Файл `outputs.tf`

```hcl
output "bucket_name" {
  value = yandex_storage_bucket.site_bucket.bucket
}

output "image_url" {
  value = "https://storage.yandexcloud.net/${yandex_storage_bucket.site_bucket.bucket}/${yandex_storage_object.image.key}"
}

output "load_balancer_ip" {
  value = yandex_lb_network_load_balancer.lamp_nlb.listener[*].external_address_spec[*].address
}

output "target_group_id" {
  value = yandex_compute_instance_group.lamp_group.load_balancer[0].target_group_id
}
```
