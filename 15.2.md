## Terraform-код


### Файл `main.tf`

    terraform {
      required_providers {
        yandex = {
          source = "yandex-cloud/yandex"
        }
      }
    }

    provider "yandex" {
      zone = "ru-central1-a"
    }

    locals {
      folder_id = "b1g7jt8jrg491cftc7qa"
    }

    data "yandex_vpc_network" "develop" {
      name = "develop"
    }

    data "yandex_vpc_subnet" "public" {
      name = "public"
    }

    resource "yandex_iam_service_account" "ig_sa" {
      name      = "tf-ig-sa"
      folder_id = local.folder_id
    }

    resource "yandex_resourcemanager_folder_iam_member" "ig_compute_editor" {
      folder_id = local.folder_id
      role      = "compute.editor"
      member    = "serviceAccount:${yandex_iam_service_account.ig_sa.id}"
    }

    resource "yandex_resourcemanager_folder_iam_member" "ig_lb_editor" {
      folder_id = local.folder_id
      role      = "load-balancer.editor"
      member    = "serviceAccount:${yandex_iam_service_account.ig_sa.id}"
    }

    resource "yandex_storage_bucket" "image_bucket" {
      bucket_prefix = "zis-"
      folder_id     = local.folder_id
      force_destroy = true

      anonymous_access_flags {
        read        = true
        list        = false
        config_read = true
      }
    }

    resource "yandex_storage_object" "image" {
      bucket       = yandex_storage_bucket.image_bucket.bucket
      key          = "image.svg"
      source       = "./files/image.svg"
      content_type = "image/svg+xml"
      acl          = "public-read"
    }

    resource "yandex_compute_instance_group" "lamp_group" {
      name               = "lamp-group"
      folder_id          = local.folder_id
      service_account_id = yandex_iam_service_account.ig_sa.id

      instance_template {
        platform_id = "standard-v3"

        resources {
          cores  = 2
          memory = 2
        }

        boot_disk {
          mode = "READ_WRITE"
          initialize_params {
            image_id = "fd827b91d99psvq5fjit"
          }
        }

        network_interface {
          network_id = data.yandex_vpc_network.develop.id
          subnet_ids = [data.yandex_vpc_subnet.public.id]
        }

        metadata = {
          "user-data" = file("${path.module}/cloud-init.yaml")
        }
      }

      scale_policy {
        fixed_scale {
          size = 3
        }
      }

      allocation_policy {
        zones = ["ru-central1-a"]
      }

      deploy_policy {
        max_unavailable = 1
        max_expansion   = 0
      }

      load_balancer {
        target_group_name        = "lamp-target-group"
        target_group_description = "Target group for homework 15.2"
      }

      depends_on = [
        yandex_resourcemanager_folder_iam_member.ig_compute_editor,
        yandex_resourcemanager_folder_iam_member.ig_lb_editor
      ]
    }

    resource "yandex_lb_network_load_balancer" "lamp_nlb" {
      name = "lamp-network-load-balancer"

      listener {
        name = "lamp-listener"
        port = 80

        external_address_spec {
          ip_version = "ipv4"
        }
      }

      attached_target_group {
        target_group_id = yandex_compute_instance_group.lamp_group.load_balancer.0.target_group_id

        healthcheck {
          name = "http"

          http_options {
            port = 80
            path = "/index.html"
          }
        }
      }
    }

    output "network_id" {
      value = data.yandex_vpc_network.develop.id
    }

    output "public_subnet_id" {
      value = data.yandex_vpc_subnet.public.id
    }

    output "bucket_name" {
      value = yandex_storage_bucket.image_bucket.bucket
    }

    output "image_url" {
      value = format(
        "https://storage.yandexcloud.net/%s/%s",
        yandex_storage_bucket.image_bucket.bucket,
        yandex_storage_object.image.key
      )
    }

    output "target_group_id" {
      value = yandex_compute_instance_group.lamp_group.load_balancer.0.target_group_id
    }

    output "load_balancer_ip" {
      value = yandex_lb_network_load_balancer.lamp_nlb.listener[0].external_address_spec[0].address
    }
